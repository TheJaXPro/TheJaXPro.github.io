    def __len__(self):
        return len(self.data)
    def __getitem__(self,index):
        return self.transforms(self.data[index]),self.target[index]

class FocalLoss(nn.Module):
   def __init__(self, class_num, alpha=None, gamma=2, size_average=True):
       super(FocalLoss, self).__init__()
       if alpha is None:
           self.alpha = Variable(torch.ones(class_num, 1))
       else:
           if isinstance(alpha, Variable):
               self.alpha = alpha
           else:
               self.alpha = Variable(alpha)
       self.gamma = gamma
       self.class_num = class_num
       self.size_average = size_average
   def forward(self, inputs, targets):
       N = inputs.size(0)
       C = inputs.size(1)
       P = F.softmax(inputs)
       class_mask = inputs.data.new(N, C).fill_(0)
       class_mask = Variable(class_mask)
       ids = targets.view(-1, 1)
       class_mask.scatter_(1, ids.data, 1.)
       self.alpha = self.alpha.cuda()
       alpha = self.alpha[ids.data.view(-1)]
       probs = (P*class_mask).sum(1).view(-1,1)
       log_p = probs.log()
       batch_loss = -alpha*(torch.pow((1-probs), self.gamma))*log_p 
       if self.size_average:
           loss = batch_loss.mean()
       else:
           loss = batch_loss.sum()
       return loss

#实现验证集和训练集准确率计算
def ver(dataloader,length):
    torch.cuda.empty_cache()
    allacc = 0
    with torch.no_grad():
        for alpht,coach in enumerate(dataloader):
            image,target = coach
            image = image.cuda()
            target = target.cuda()
            pred = model(image)
            index = torch.max(pred,1)[1]
            acc = torch.sum(torch.eq(index,target)).item()
            allacc = allacc + acc
    torch.cuda.empty_cache()
    return allacc/length

    maxdata = data.max()
    mindata = data.min()
    k = 255/(maxdata-mindata)
