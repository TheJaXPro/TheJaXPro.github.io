实现判别器¶

基本实现思路就是对于K个label，分别用K个一维正态分布去表示，然后对于K个正态分布分别求参数mu和sig(MLE,MAP或者是贝叶斯方法求概率密度都可以)，就得到了分类判别器

最后运用贝叶斯公式就可以求解新样本的预测值

#实现正态分布的概率求解
def normal(data,mu,sig):
    return (1/np.sqrt(2*np.pi)*sig)*np.exp(-1*np.power(data - mu,2)/2*(sig**2))
#实现函数用最大似然法求解正态分布的mu和sig
def MLE(data):
    mu = np.mean(data)
    sig = np.mean(np.power(data - mu,2))
    return mu,sig
    
#定义数据，分为两类,小于0的和大于0的
data_zeros = np.random.normal(-5,1,200)
data_ones = np.random.normal(5,1,200)

#实现对于两种不同的正态分布进行最大似然法
mu_zeros,sig_zeors = MLE(data_zeros)
mu_ones,sig_ones = MLE(data_ones)

x = np.linspace(-10,10,1200)

label0 = normal(x,mu_zeros,sig_zeors)
label1 = normal(x,mu_ones,sig_ones)
#前面求P(x|w),这里用贝叶斯公式求P(w|x)，这里的运算可以直接简化成如下相除形式
zeros = label0/(label0+label1)
ones = label1/(label0+label1)

#画图
plt.figure(figsize=(16,16))
plt.subplot(211)
plt.plot(x,label0,label = "zeors classifier")
plt.plot(x,label1,label = "ones classifier")
plt.xlabel("MLE function")
plt.legend(loc = 'best')

plt.subplot(212)
plt.plot(x,zeros,label = "label of zeors possibilty")
plt.plot(x,ones,label = "label of ones possibilty")
plt.xlabel("posibillty of ones or zeros")
plt.legend(loc = 'best')
plt.show()
