def normalpdf(mu : int,sig : int,x : int) -> float:
    return np.exp(np.power(x - mu,2)/(-2*np.power(sig,2)))/(np.sqrt(2*np.pi)*sig)

#定义MLE过程的函数
def mu_sig(arr :"np.array") -> [int,int]:
    """
    Paramer:
        arr: array of data
    return:
        mu: the mean of arr
        sig:the variance of the arr
        
        the mu and sig is the result of the Maximum likelihood 
    """
    mu = np.mean(arr)
    sig = np.sum(np.power(arr - mu,2))/len(arr)
    return mu,sig

#定义函数生成正态分布的概率值
MLE_mu,MLE_sig = mu_sig(data)
origin_re = [normalpdf(origin_mu,origin_sig,i) for i in x]
MLE_re = [normalpdf(MLE_mu,MLE_sig,i) for i in x]

最大后验法 MAP1¶

使得贝叶斯公式后得后验概率和先验概率得形式不发生变化的似然函数，我们称这两个不同的概率分布为共轭，这可以使得我们在进行贝叶斯变换的过程不发生后验概率模型形式上的变化，所以在MAP的过程中，一般我们会选择和似然函数相共轭的概率分布来表示参数的概率分布，这都是为了方便我们进行运算

根据MAP的公式，对比起MLE的公式，就是在似然函数的基础上乘上一个参数的概率值，如果参数的概率值为均匀分布，那么MAP和MLE是一样的，所以MLE可以看作是MAP的特殊情况。

那么同样对于一维正态分布而言，我们选择正态逆伽马分布作为参数的概率分布，同样累乘(或者累乘后取对数值)，再对参数mu，sig求偏导，并令其等于零，可以得到表达式有：

#定义MAP过程的函数
def MAP_normal(x : "np.array",alpha = 1,beta = 1,gamma = 1,delta = -1) -> [int,int]:
    """
    Paramer:
            alpha = 1,beta = 1,gamma = 1,delta = -1: the parameters of Normal inverse gamma distribution
            x:data of x_train
    return:
            the mu and sig of MAP
    """
    
    mu = (np.sum(x)+gamma*delta)/(len(x)+gamma)
    sig = (np.sum(np.power(x - mu,2))+2*beta+gamma*(delta - mu))/(len(x)+3+2*alpha)
    
    return mu,sig

MAP_mu,MAP_sig = MAP_normal(data)

origin_re = [normalpdf(origin_mu,origin_sig,i) for i in x]
MAP_re = [normalpdf(MAP_mu,MAP_sig,i) for i in x]

def Byes(x_train:"np.array",x_test:"np.array",alpha = 1,beta = 1,gamma = 1,delta = 0) ->"np.array":
    """
    Paramer:
            alpha = 1,beta = 1,gamma = 1,delta =0 : the Previous parameters of Normal inverse gamma distribution
            x_train                               : the data to Update parameters of Normal inverse gamma distribution
            x_test                                : the data to prediction
    return:
            newParamer: alpha, beta, gamma, delta of NewGenerator
    """
    I = len(x_train)
    new_alpha = alpha + 0.5*I
    new_gamma = gamma + I
    new_delat = (gamma*delta+np.sum(x_train))/(gamma+I)
    new_beta = np.sum(np.power(x_train,2))/2 + beta+ gamma*(delta**2)/2 - np.power(gamma*delta+np.sum(x_train),2)/2*(gamma+I)

    alpha_p = new_alpha + 0.5
    gamma_p = new_gamma + 1
    beta_p = np.power(x_test,2)/2 + new_beta + new_gamma*(new_delat**2)/2 - np.power(new_gamma*new_delat+x_test,2)/2*(new_gamma+1)
    print(new_alpha,alpha_p)
    temp = math.gamma(alpha_p)/math.gamma(new_alpha)
    temp = temp*np.sqrt(new_gamma/gamma_p)*(new_beta**new_alpha)*(1/np.sqrt(2*np.pi))

    result = np.ones((1,len(x_test)))*temp
    down = beta_p**alpha_p                                                                               
    
    result = result/down
    
    return result
